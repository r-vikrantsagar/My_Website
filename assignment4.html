### 1. Table with Embedded Charts

Let's use font 'Times' and colors 'blueviolet' and 'mediumpurple1'.

```{r}
hpi <- subset(hpi, select=-...4) # drop ...4 column
names(hpi) = c('rank', 'country', 'iso', 'continent', 'population', 'life_exp', 'wellbeing', 'footprint', 'hpi', 'biocapacity', 'gdp') # rename columns
class(hpi$gdp) = "double" # set gdp from character to double

#classic plot without fonts and color
# ggplot(hpi) + 
#   geom_dotplot(aes(x=hpi)) + 
#   facet_wrap(vars(continent))

#plot with font and colors
ggplot(hpi) + 
  geom_dotplot(aes(x=hpi),
               color = "blueviolet",
               fill = "mediumpurple1") + 
  facet_wrap(vars(continent)) + 
  theme(text = element_text(family="Times")) +
  labs(title = "Human Planet Index",
       subtitle = "Year of 2022",
       caption = "HPI by continent",
       tag = "Figure 1",
       x = "HPI",
       y = "Number of countries",
       colour = "GrandBudapest")
```

### 2. Variable Width Column Chart

Let's use font 'Times' and colors 'gray0' and 'tomato1'.

```{r}
test <- read_xlsx("test_data.xlsx")
summary(test)

#classic plot without fonts and color
# ggplot(test) + 
#   geom_histogram(aes(x=height), 
#                  color='steelblue4', 
#                  fill='steelblue3', 
#                  breaks=unique(test$weight))

#plot with font and colors
ggplot(test) + 
  geom_histogram(aes(x=height), 
                 color='gray0', 
                 fill='tomato1', 
                 breaks=unique(test$weight)) +
  theme(text = element_text(family="Times")) +
  labs(title = "Test Data",
       subtitle = "Year of 2022",
       caption = "Dummy data",
       tag = "Figure 2",
       x = "Height",
       y = "Number of people")
```
